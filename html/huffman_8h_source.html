95int calculate_bits_trashed(PRIORITY_QUEUE* pq, HuffmanCode huff_table[256]) {
96 int bit_amount = 0;
97 while (pq->size > 0) {
98 NODE* node = remove_lower(pq);
99 if (node && huff_table[node->character].length > 0) {
100 // bits = (frequência * comprimento do código)
101 bit_amount += node->frequency * huff_table[node->character].length;
102 }
103 }
104
105 return bit_amount;

159void write_header(PRIORITY_QUEUE* pq, HuffmanCode huff_table[256], FILE *output_file, NODE* root) {
160
161 int total_bits = calculate_bits_trashed(pq, huff_table);
162 int trash = ((8 - (total_bits % 8)) % 8);
163 int tree_size = count_tree_size(root);
164
165 // Empacota o lixo (3 bits) e o tamanho da árvore (13 bits) em 16 bits.
166 unsigned short header = (trash << 13) | tree_size;
167
168 // Divide o header em dois bytes para escrita.
169 unsigned char byte1 = header >> 8;
170 unsigned char byte2 = header & 0xFF;
171
172 fwrite(&byte1, 1, 1, output_file);
173 fwrite(&byte2, 1, 1, output_file);
174
175 write_tree(root, output_file);
176}

230void compactor(FILE *input_file, FILE *output_file, HuffmanCode huff_table[256]) {
231 unsigned char c;
232 BitBuffer bit_buffer = {0, 0};
233
234 while (fread(&c, 1, 1, input_file) == 1) {
235 HuffmanCode code = huff_table[c];
236
237 for (int i = code.length - 1; i >= 0; i--) {
238 bit_buffer_add(&bit_buffer, (code.code >> i) & 1);
239
240 if (bit_buffer.bits_used == 8) {
241 write_buffer(output_file, &bit_buffer);
242 }
243 }
244 }
245
246 // Escreve os bits restantes (com lixo) no final do arquivo.
247 write_buffer(output_file, &bit_buffer);
248}

365void decompress(FILE *input, FILE *output, NODE* root, int trash_size, int header_bytes) {
366 fseek(input, 0, SEEK_END);
367 long file_size = ftell(input);
368 long data_size = file_size - header_bytes;
369 fseek(input, header_bytes, SEEK_SET);
370
371 NODE* current = root;
372 unsigned char byte;
373 for (long i = 0; i < data_size; i++) {
374 fread(&byte, 1, 1, input);
375
376 for (int bit = 7; bit >= 0; bit--) {
377 int current_bit = (byte >> bit) & 1;
378 if (current_bit == 0)
379 current = current->left;
380 else
381 current = current->right;
382
383 if (is_leaf(current)) {
384 fputc(current->character, output);
385 current = root;
386 }
387
388 if (i == data_size - 1 && bit == trash_size - 1) {
389 return;
390 }
391 }
392 }
